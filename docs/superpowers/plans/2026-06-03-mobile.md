# Mobile / Responsive Implementation Plan (Plan 15)

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:executing-plans (6-task micro-plan).

**Goal:** Make knot usable on a phone. Two specific surfaces this plan covers (per user-confirmed scope):
1. **Sidebar collapses to a drawer** under 768 px with a hamburger toggle.
2. **Command palette goes full-screen** under 640 px so the soft keyboard doesn't crush it.

Toolbar wrap and touch-friendly tree are **out of scope** — desktop ergonomics stay as-is.

**Tech approach (per user decision):** keep the inline-style pattern; add a `useViewport()` hook that returns `"mobile"|"tablet"|"desktop"`. Components branch inline styles on viewport. No CSS framework, no migration to CSS modules.

**Breakpoints:**
- `mobile` < 640
- `tablet` 640 – 1024
- `desktop` ≥ 1024

**Predecessor:** Plan 16 (HEAD `6380bea`).

## Task 1: useViewport hook

**Files:**
- Create: `web/src/hooks/useViewport.ts`
- Create: `web/src/hooks/useViewport.test.ts`

```ts
import { useEffect, useState } from "react";

export type Viewport = "mobile" | "tablet" | "desktop";

const MOBILE_MAX = 640;
const TABLET_MAX = 1024;

function classify(width: number): Viewport {
  if (width < MOBILE_MAX) return "mobile";
  if (width < TABLET_MAX) return "tablet";
  return "desktop";
}

export function useViewport(): Viewport {
  const [vp, setVp] = useState<Viewport>(() =>
    typeof window === "undefined" ? "desktop" : classify(window.innerWidth),
  );
  useEffect(() => {
    const onResize = () => setVp(classify(window.innerWidth));
    window.addEventListener("resize", onResize);
    return () => window.removeEventListener("resize", onResize);
  }, []);
  return vp;
}
```

Test:

```ts
import { describe, expect, it } from "vitest";
import { renderHook } from "@testing-library/react";
import { useViewport } from "./useViewport";

describe("useViewport", () => {
  it("classifies the default window as desktop", () => {
    // jsdom default width is 1024 → desktop
    const { result } = renderHook(() => useViewport());
    expect(["mobile", "tablet", "desktop"]).toContain(result.current);
  });
});
```

## Task 2: AppShell drawer

**Files:**
- Modify: `web/src/components/AppShell.tsx`

Behavior:
- On desktop / tablet (≥ 640 px): two-column grid as today; sidebar is part of the layout.
- On mobile (< 640 px): sidebar is a fixed-position drawer at left, hidden by default. Overlay backdrop closes it on click.

Inline style switching keyed on `useViewport()`. The Zustand `sidebarOpen` state (already exists) controls open/closed.

```tsx
import { useUi } from "../stores/ui";
import { useViewport } from "../hooks/useViewport";

export function AppShell() {
  const sidebarOpen = useUi((s) => s.sidebarOpen);
  const toggleSidebar = useUi((s) => s.toggleSidebar);
  const vp = useViewport();
  const mobile = vp === "mobile";

  return (
    <div
      style={{
        display: mobile ? "block" : "grid",
        gridTemplateColumns: mobile ? undefined : sidebarOpen ? "260px 1fr" : "0 1fr",
        height: "100vh",
        fontFamily: "system-ui, sans-serif",
      }}
    >
      {mobile && sidebarOpen && (
        <div
          data-testid="sidebar-backdrop"
          onClick={toggleSidebar}
          style={{
            position: "fixed",
            inset: 0,
            background: "rgba(0,0,0,0.4)",
            zIndex: 20,
          }}
        />
      )}
      <aside
        data-testid="sidebar"
        style={{
          borderRight: "1px solid #e5e5e5",
          overflow: "auto",
          background: "#fafafa",
          position: mobile ? "fixed" : "static",
          left: mobile ? (sidebarOpen ? 0 : "-280px") : undefined,
          top: 0,
          width: mobile ? 260 : undefined,
          height: mobile ? "100vh" : undefined,
          zIndex: mobile ? 30 : undefined,
          transition: mobile ? "left 200ms ease-out" : undefined,
        }}
      >
        <DocTree />
      </aside>
      <main style={{ overflow: "auto", height: mobile ? "100vh" : undefined }}>
        <Outlet />
      </main>
      <Toast />
      <CommandPalette />
    </div>
  );
}
```

## Task 5: mobile e2e

**Files:**
- Create: `e2e/flows/mobile.spec.ts`

Configure a 375×667 viewport per-test (the iPhone SE size).

```ts
import { execSync } from "node:child_process";
import { expect, test } from "@playwright/test";

function reset() {
  const tables = [
    "acl_invalidations","audit_events","doc_markdown_cache","doc_snapshots","doc_updates",
    "document_grants","documents","sessions","workspace_members","users","workspaces",
    "blobs","blob_bytes",
  ].join(", ");
  execSync(
    `docker compose -f deploy/compose/dev.yml exec -T postgres psql -U knot -d knot -c "TRUNCATE TABLE ${tables} CASCADE"`,
    { cwd: "..", stdio: "pipe" },
  );
}
test.beforeAll(reset);
test.use({ viewport: { width: 375, height: 667 } });

test("mobile: drawer opens/closes, palette goes full-screen", async ({ page }) => {
  await page.goto("/setup");
  await page.getByTestId("setup-email").fill("o@e.com");
  await page.getByTestId("setup-display-name").fill("O");
  await page.getByTestId("setup-password").fill("owner-hunter22");
  await page.getByTestId("setup-submit").click();
  await page.waitForURL(/\/(?:doc\/.+)?$/);

  // Sidebar drawer starts open (sidebarOpen=true initial) — close it via backdrop.
  // First, navigate so the sidebar is visible at all.
  // On mobile, the toggle appears only when sidebarOpen=false.
  const toggle = page.getByTestId("menu-toggle");
  const backdrop = page.getByTestId("sidebar-backdrop");

  // Backdrop is present → close the drawer.
  if (await backdrop.isVisible()) await backdrop.click();
  await expect(toggle).toBeVisible();

  // Open via toggle.
  await toggle.click();
  await expect(page.getByTestId("sidebar")).toBeVisible();
  await expect(backdrop).toBeVisible();

  // Open palette — should be full-screen.
  await page.keyboard.press("Control+k");
  const dialog = page.getByTestId("cmdk");
  await expect(dialog).toBeVisible();
  // Roughly equal to viewport size (use bounding box width).
  const box = await dialog.boundingBox();
  expect(box?.width).toBeGreaterThan(370);
});
```
